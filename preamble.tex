% Звёздочка для рейтинга
\newcommand{\ratingformat}[1]{$\bigstar$ #1}

% Команда для задачи: \problem{номер}{рейтинг}
\newcommand{\problem}[2]{%
    \par\medskip
    \noindent\textbf{#1} \quad \textcolor{ratingcolor}{\ratingformat{#2}}
    \par\medskip
}

% Команда для ответа
\newcommand{\answer}[2]{%
    \par\medskip
    \noindent\textbf{#1. Ответ:} #2
    \par\smallskip
}

% Команда для решения (опционально)
\newcommand{\solution}[1]{%
    \par\smallskip
    \noindent\textit{Решение:} #1
    \par\medskip
}
