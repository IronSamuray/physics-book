% === preamble.tex ===
% Минималистичные настройки оформления сборника задач

% --- Кодировка и язык ---
\usepackage[T2A]{fontenc}
\usepackage[utf8]{inputenc}
\usepackage[russian]{babel}

% --- Математика ---
\usepackage{amsmath, amssymb}

% --- Картинки ---
\usepackage{graphicx}

% --- Ссылки и оглавление ---
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=blue,
    urlcolor=cyan,
    pdfauthor={IronSamuray},
    pdftitle={Сборник задач по физике 10-11 класс}
}

% --- Макет страницы ---
\usepackage{geometry}
\geometry{a4paper, top=2cm, bottom=2cm, left=2cm, right=2cm}

% --- Цвета ---
\usepackage{xcolor}
\definecolor{ratingcolor}{gray}{0.5} % Серый цвет для рейтинга

% === ШАБЛОНЫ ДЛЯ ЗАДАЧ ===

% Переключатель решений
\newif\ifsolutions
\solutionstrue  % Показывать решения (по умолчанию)

% Звёздочка для рейтинга (выбирай какой нравится):
% \star - простая звезда
% \bigstar - жирная звезда  
% \ast - астериск *
\newcommand{\ratingformat}[1]{$\bigstar$ #1}

% Команда для оформления задачи
% Использование: \problem{номер}{рейтинг}
\newcommand{\problem}[2]{%
    \par\medskip
    \noindent\textbf{Задача №#1} \quad \textcolor{ratingcolor}{\ratingformat{#2}}
    \par\nopagebreak
    \smallskip
}

% Окружение для решения
\newenvironment{solution}{%
    \ifsolutions
    \par\medskip
    \noindent\textbf{Решение:}
    \par\smallskip
}{%
    \par\medskip
    \fi
}

% === КОМАНДЫ ДЛЯ КАРТИНОК ===

% Картинка с подписью
% Использование: \fig{0.5}{images/mechanics/plot.png}{Подпись}
\newcommand{\fig}[3]{
    \begin{center}
        \includegraphics[width=#1\textwidth]{#2}
        \par\medskip
        \textbf{Рис.} #3
    \end{center}
}

% Картинка без подписи
% Использование: \image{0.5}{images/mechanics/plot.png}
\newcommand{\image}[2]{
    \begin{center}
        \includegraphics[width=#1\textwidth]{#2}
    \end{center}
}
